on_title_gain = {
	on_actions = {
		on_start_coronation
	}
}

on_start_coronation = {
	trigger = {
		NOT = { has_character_flag = stopping_firing_please }
	}
	effect = {
		trigger_event = {
			id = cmh_coronations.002
			days = { 1 60 }
		}
	}
}

on_title_lost = {
	on_actions = {
		on_clear_coronation
	}
}

on_clear_coronation = {
	trigger = {
		NAND = {
			highest_held_title_tier > 2
			is_independent_ruler = yes
		}
	}
	effect = {
		clear_uncoronated_debuff = yes
	}
}
